```python
CRITICAL_RISK_THRESHOLD = 15  # net_score >= this = critical

def is_critical_risk(risk: "Risk") -> bool:
    """Check if a risk is critical (requires approval for linked item edits)."""
    if risk.is_priority:
        return True
    if risk.net_score >= CRITICAL_RISK_THRESHOLD:
        return True
    return False

SENSITIVE_FIELDS = {
    "risk": {"owner_id", "department_id", "category", "is_priority"},
    "control": {"control_owner_id", "department_id"},
    "kri": {},  # KRIs inherit sensitivity from linked risk
}

def has_sensitive_field_changes(resource_type: str, old_data: dict, new_data: dict) -> tuple[bool, dict]:
    """
    Check if any sensitive fields are being changed.
    
    Returns: (has_sensitive, changed_fields_dict)
    """
    sensitive = SENSITIVE_FIELDS.get(resource_type, set())
    changed = {}
    
    for field in sensitive:
        old_val = old_data.get(field)
        new_val = new_data.get(field)
        if new_val is not None and old_val != new_val:
            # Special case: is_priority can only go false→true without approval
            if field == "is_priority":
                if old_val == True and new_val == False:
                    changed[field] = {"old": old_val, "new": new_val}
            else:
                changed[field] = {"old": old_val, "new": new_val}
    
    return bool(changed), changed
```

async def update_risk(
    risk_id: int,
    risk_data: RiskUpdate,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(deps.get_current_user),
):
    # ... existing validation ...
    
    from app.core.permissions import can_resolve_approvals, has_sensitive_field_changes
    from app.models import ApprovalRequest, ApprovalStatus, ApprovalResourceType, ApprovalActionType
    import json
    
    # Skip approval check for privileged users
    if not can_resolve_approvals(current_user):
        # Check for sensitive field changes
        old_data = {"owner_id": risk.owner_id, "department_id": risk.department_id, 
                    "category": risk.category, "is_priority": risk.is_priority}
        update_dict = risk_data.model_dump(exclude_unset=True)
        
        has_sensitive, changed = has_sensitive_field_changes("risk", old_data, update_dict)
        
        if has_sensitive:
            # Create approval request instead of applying changes
            approval = ApprovalRequest(
                resource_type=ApprovalResourceType.RISK,
                resource_id=risk.id,
                resource_name=f"{risk.risk_id_code}: {risk.description[:50] if risk.description else ''}",
                requested_by_id=current_user.id,
                reason=f"Requested change to sensitive fields: {', '.join(changed.keys())}",
                action_type=ApprovalActionType.EDIT,
                pending_changes=json.dumps(changed),
                status=ApprovalStatus.PENDING,
            )
            db.add(approval)
            await db.commit()
            
            raise HTTPException(
                status_code=status.HTTP_202_ACCEPTED,
                detail={"message": "Change requires approval", "approval_id": approval.id, "pending_fields": list(changed.keys())}
            )
    
    # Apply non-sensitive changes (or all changes if privileged)
    # ... rest of existing update logic ...
