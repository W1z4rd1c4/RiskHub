### 1. Add E2E Test for Risk Manager Approval Access

**File**: `frontend/e2e/approval-role-access.spec.ts`

```typescript
import { test, expect } from '@playwright/test';

test.describe('Approval Role Access', () => {
    test('Risk Manager can see pending approvals', async ({ page }) => {
        // Login as Risk Manager (Petra Svobodová)
        await page.goto('/');
        await page.locator('[data-testid="user-select"]').click();
        await page.locator('text=Petra Svobodová').click();
        
        // Navigate to Workflow page
        await page.goto('/approvals');
        
        // Verify Pending Queue tab is active and shows items
        await expect(page.locator('[data-testid="pending-queue-tab"]')).toHaveClass(/active/);
        
        // Should NOT show "All Caught Up" if there are pending approvals
        const emptyState = page.locator('text=All Caught Up');
        const approvalItems = page.locator('[data-testid="approval-item"]');
        
        // Either there are items OR empty state (but RM should see all pending)
        const itemCount = await approvalItems.count();
        if (itemCount === 0) {
            // If no items, that's OK - but verify badge is also 0
            const badge = page.locator('[data-testid="workflow-badge"]');
            const badgeText = await badge.textContent();
            expect(badgeText).toBe('0');
        } else {
            // If items exist, verify we can see them
            expect(itemCount).toBeGreaterThan(0);
        }
    });

    test('Risk Manager can approve a request', async ({ page }) => {
        await page.goto('/');
        await page.locator('[data-testid="user-select"]').click();
        await page.locator('text=Petra Svobodová').click();
        
        await page.goto('/approvals');
        
        // Find first pending approval
        const firstApproval = page.locator('[data-testid="approval-item"]').first();
        if (await firstApproval.isVisible()) {
            // Click approve button
            await firstApproval.locator('[data-testid="approve-btn"]').click();
            
            // Confirm in modal
            await page.locator('[data-testid="confirm-approve"]').click();
            
            // Verify success toast
            await expect(page.locator('text=approved')).toBeVisible();
        }
    });

    test('Department Head cannot see all pending approvals', async ({ page }) => {
        // Login as Department Head
        await page.goto('/');
        await page.locator('[data-testid="user-select"]').click();
        await page.locator('text=Jan Novák').click(); // Dept Head
        
        await page.goto('/approvals');
        
        // Should only see their own submitted requests or ones they're primary approver for
        // Should NOT see global pending queue
    });
});
```
