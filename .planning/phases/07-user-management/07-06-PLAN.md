#### 1.2 Fix Empty `dept_ids` Permission Bypass

**File**: `backend/app/core/permissions.py`

```python
def get_user_department_ids(user: User) -> list[int] | None:
    """
    Get list of department IDs user can access.
    
    Returns:
        None: User is privileged (can see all)
        []: User has no departments (can see nothing)
        [1, 2, ...]: User can see these departments
    """
    if is_privileged_user(user):
        return None  # Privileged users see all
    
    if user.department_id:
        return [user.department_id]
    
    return []  # No department = no access

def is_privileged_user(user: User) -> bool:
    """Check if user has privileged role (full system access)."""
    if not user.role:
        return False  # Guard against null role
    return user.role.name in RoleType.privileged_roles()
```

**Update all endpoints** to use:
```python
dept_ids = get_user_department_ids(current_user)
if dept_ids is not None:
    if not dept_ids:
        return []  # or empty response
    query = query.filter(Model.department_id.in_(dept_ids))
```

### Phase 2: High Priority Fixes (Security)

#### 2.1 Add Department Scoping to CRUD Endpoints

**Pattern for all endpoints:**

```python
async def check_department_access(
    item_dept_id: int | None,
    current_user: User,
) -> None:
    """Raise 403 if user cannot access this department."""
    if item_dept_id is None:
        return  # No department = anyone can access
    
    dept_ids = get_user_department_ids(current_user)
    if dept_ids is None:
        return  # Privileged user
    
    if item_dept_id not in dept_ids:
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail="Access denied to this department's resources"
        )
```
