### 1. Add Permission Filtering to Risk Endpoints

**File**: `backend/app/api/v1/endpoints/risks.py`

Add to list endpoint:

```python
from app.api import deps
from app.core.permissions import get_user_department_ids

@router.get("", response_model=dict)
async def list_risks(
    skip: int = Query(0, ge=0),
    limit: int = Query(100, ge=1, le=100),
    current_user: User = Depends(deps.get_current_user),  # ADD THIS
    db: Session = Depends(get_db)
):
    """List risks with permission filtering."""
    query = db.query(Risk)
    
    # Apply department filtering
    dept_ids = get_user_department_ids(current_user)
    if dept_ids:  # If not empty, filter by departments
        query = query.filter(Risk.department_id.in_(dept_ids))
    
    # Rest of existing logic...
    total = query.count()
    risks = query.offset(skip).limit(limit).all()
    
    return {
        "items": [RiskSummary.from_orm(r) for r in risks],
        "total": total,
        "skip": skip,
        "limit": limit
    }
```

Apply same pattern to:
- `GET /risks/{id}` - Check user can see this risk's department
- `POST /risks` - Set created_by to current_user.id
- `PUT /risks/{id}` - Check permission before update
- `DELETE /risks/{id}` - Check permission before delete

### 6. Create Permission Hooks

**File**: `frontend/src/hooks/usePermissions.ts`

```typescript
import { useAuth } from '@/contexts/AuthContext';

export function usePermissions() {
  const { user, hasPermission } = useAuth();

  return {
    canManageUsers: hasPermission('users', 'write'),
    canCreateRisks: hasPermission('risks', 'write'),
    canDeleteRisks: hasPermission('risks', 'delete'),
    canCreateControls: hasPermission('controls', 'write'),
    canDeleteControls: hasPermission('controls', 'delete'),
    canViewReports: hasPermission('reports', 'read'),
    isPrivilegedUser: user?.role in ['ceo', 'cfo', 'cro', 'risk_manager', 'compliance', 'legal', 'internal_audit', 'actuarial', 'admin'],
    isDepartmentHead: user?.role === 'department_head',
    isEmployee: user?.role === 'employee',
  };
}
```

```python
def test_privileged_user_sees_all_risks(client, cro_user, auth_headers_cro):
    """CRO should see all risks across all departments."""
    response = client.get("/api/v1/risks", headers=auth_headers_cro)
    assert response.status_code == 200
    data = response.json()
    assert data["total"] == 50  # All risks

def test_department_head_sees_only_department_risks(client, coo_user, auth_headers_coo):
    """COO should see only Operations department risks."""
    response = client.get("/api/v1/risks", headers=auth_headers_coo)
    assert response.status_code == 200
    data = response.json()
    assert data["total"] == 10  # Only Operations risks
    assert all(r["department_name"] == "Operations" for r in data["items"])

def test_employee_sees_only_department_risks(client, employee_user, auth_headers_employee):
    """Employee should see only their department's risks."""
    response = client.get("/api/v1/risks", headers=auth_headers_employee)
    assert response.status_code == 200
    data = response.json()
    assert all(r["department_id"] == employee_user.department_id for r in data["items"])
```
