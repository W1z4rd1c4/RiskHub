### 1. Create Roles and Permissions Seed Script

**File**: `backend/scripts/seed_roles_permissions.py`

```python
from sqlalchemy.orm import Session
from app.db.session import SessionLocal
from app.models import Role, Permission, RolePermission

def seed_roles_permissions():
    db = SessionLocal()
    
    # Create permissions
    permissions = [
        Permission(resource="*", action="*", description="Full access"),
        Permission(resource="risks", action="read", description="View risks"),
        Permission(resource="risks", action="write", description="Create/edit risks"),
        Permission(resource="risks", action="delete", description="Delete risks"),
        Permission(resource="controls", action="read", description="View controls"),
        Permission(resource="controls", action="write", description="Create/edit controls"),
        Permission(resource="controls", action="delete", description="Delete controls"),
        Permission(resource="kris", action="read", description="View KRIs"),
        Permission(resource="kris", action="write", description="Create/edit KRIs"),
        Permission(resource="reports", action="read", description="View reports"),
        Permission(resource="users", action="read", description="View users"),
        Permission(resource="users", action="write", description="Manage users"),
    ]
    
    for perm in permissions:
        existing = db.query(Permission).filter(
            Permission.resource == perm.resource,
            Permission.action == perm.action
        ).first()
        if not existing:
            db.add(perm)
    
    db.commit()
    
    # Create roles
    roles_config = [
        ("admin", "Administrator", "Full system access"),
        ("ceo", "Chief Executive Officer", "C-Suite executive"),
        ("cfo", "Chief Financial Officer", "C-Suite executive"),
        ("cro", "Chief Risk Officer", "C-Suite executive"),
        ("coo", "Chief Operating Officer", "C-Suite executive"),
        ("risk_manager", "Risk Manager", "Risk management oversight"),
        ("compliance", "Compliance Officer", "Compliance oversight"),
        ("legal", "Legal Counsel", "Legal oversight"),
        ("internal_audit", "Internal Auditor", "Audit oversight"),
        ("actuarial", "Actuarial Function", "Actuarial oversight"),
        ("department_head", "Department Head", "Department manager"),
        ("employee", "Employee", "Standard employee"),
        ("viewer", "Viewer", "Read-only access"),
    ]
    
    for name, display_name, description in roles_config:
        existing = db.query(Role).filter(Role.name == name).first()
        if not existing:
            role = Role(name=name, display_name=display_name, description=description)
            db.add(role)
    
    db.commit()
    
    # Assign permissions to roles
    full_access_perm = db.query(Permission).filter(
        Permission.resource == "*", Permission.action == "*"
    ).first()
    
    privileged_roles = ["admin", "ceo", "cfo", "cro", "risk_manager", 
                       "compliance", "legal", "internal_audit", "actuarial"]
    
    for role_name in privileged_roles:
        role = db.query(Role).filter(Role.name == role_name).first()
        if role and full_access_perm:
            existing = db.query(RolePermission).filter(
                RolePermission.role_id == role.id,
                RolePermission.permission_id == full_access_perm.id
            ).first()
            if not existing:
                db.add(RolePermission(role_id=role.id, permission_id=full_access_perm.id))
    
    # Department heads and employees get limited permissions
    read_write_perms = db.query(Permission).filter(
        Permission.action.in_(["read", "write"])
    ).all()
    
    for role_name in ["department_head", "employee"]:
        role = db.query(Role).filter(Role.name == role_name).first()
        if role:
            for perm in read_write_perms:
                if perm.resource in ["risks", "controls", "kris"]:
                    existing = db.query(RolePermission).filter(
                        RolePermission.role_id == role.id,
                        RolePermission.permission_id == perm.id
                    ).first()
                    if not existing:
                        db.add(RolePermission(role_id=role.id, permission_id=perm.id))
    
    db.commit()
    db.close()
    print("✅ Roles and permissions seeded")

if __name__ == "__main__":
    seed_roles_permissions()
```

### 2. Create Users Seed Script

**File**: `backend/scripts/seed_users.py`

```python
from sqlalchemy.orm import Session
from app.db.session import SessionLocal
from app.models import User, Role, Department
from app.core.security import get_password_hash
import random

def seed_users():
    db = SessionLocal()
    
    # Get roles
    roles = {r.name: r for r in db.query(Role).all()}
    
    # Get departments
    departments = {d.name: d for d in db.query(Department).all()}
    
    users_to_create = []
    
    # 1. C-Suite (4 users)
    c_suite = [
        ("ceo@riskhub.test", "Maria Silva", "ceo", None),
        ("cfo@riskhub.test", "John Chen", "cfo", None),
        ("cro@riskhub.test", "Anna Kowalski", "cro", None),  # DEMO ACCOUNT
        ("coo@riskhub.test", "Robert Johnson", "coo", "Operations"),  # DEMO ACCOUNT
    ]
    
    for email, name, role_name, dept_name in c_suite:
        users_to_create.append({
            "email": email,
            "name": name,
            "role_id": roles[role_name].id,
            "department_id": departments[dept_name].id if dept_name else None,
            "manager_id": None,
            "hashed_password": get_password_hash("test123")
        })
    
    # 2. Governance Roles (6 users)
    governance = [
        ("risk.manager@riskhub.test", "Sarah Williams", "risk_manager", "Risk Management"),
        ("compliance@riskhub.test", "Michael Brown", "compliance", "Compliance"),
        ("legal@riskhub.test", "Emma Davis", "legal", "Legal"),
        ("audit@riskhub.test", "James Wilson", "internal_audit", None),
        ("actuarial@riskhub.test", "Lisa Anderson", "actuarial", "Actuarial"),
        ("admin@riskhub.test", "System Admin", "admin", None),
    ]
    
    for email, name, role_name, dept_name in governance:
        users_to_create.append({
            "email": email,
            "name": name,
            "role_id": roles[role_name].id,
            "department_id": departments[dept_name].id if dept_name else None,
            "manager_id": None,
            "hashed_password": get_password_hash("test123")
        })
    
    # 3. Department Heads (10 users, one per department)
    dept_heads = []
    first_names = ["Alex", "Jordan", "Taylor", "Morgan", "Casey", "Riley", "Avery", "Quinn", "Sage", "River"]
    last_names = ["Smith", "Johnson", "Williams", "Brown", "Jones", "Garcia", "Miller", "Davis", "Rodriguez", "Martinez"]
    
    for i, (dept_name, dept) in enumerate(departments.items()):
        if dept_name == "Operations":
            continue  # COO is already the head
        
        email = f"{dept_name.lower().replace(' ', '.')}.head@riskhub.test"
        name = f"{first_names[i % len(first_names)]} {last_names[i % len(last_names)]}"
        
        dept_heads.append({
            "email": email,
            "name": name,
            "role_id": roles["department_head"].id,
            "department_id": dept.id,
            "manager_id": None,  # Will be set to CEO later
            "hashed_password": get_password_hash("test123")
        })
    
    users_to_create.extend(dept_heads)
    
    # Create users in database to get IDs
    db_users = {}
    for user_data in users_to_create:
        existing = db.query(User).filter(User.email == user_data["email"]).first()
        if not existing:
            user = User(**user_data)
            db.add(user)
            db.flush()
            db_users[user_data["email"]] = user
        else:
            db_users[user_data["email"]] = existing
    
    db.commit()
    
    # 4. Employees (100 users across departments)
    # Each department gets 8-12 employees
    employee_count = 0
    first_names_pool = ["Emma", "Liam", "Olivia", "Noah", "Ava", "Ethan", "Sophia", "Mason", "Isabella", "William",
                        "Mia", "James", "Charlotte", "Benjamin", "Amelia", "Lucas", "Harper", "Henry", "Evelyn", "Alexander"]
    last_names_pool = ["Smith", "Johnson", "Williams", "Brown", "Jones", "Garcia", "Miller", "Davis", "Rodriguez", "Martinez",
                       "Hernandez", "Lopez", "Gonzalez", "Wilson", "Anderson", "Thomas", "Taylor", "Moore", "Jackson", "Martin"]
    
    for dept_name, dept in departments.items():
        # Find department head
        dept_head = db.query(User).filter(
            User.department_id == dept.id,
            User.role_id == roles["department_head"].id
        ).first()
        
        if not dept_head and dept_name == "Operations":
            dept_head = db_users["coo@riskhub.test"]
        
        # Create 8-12 employees per department
        num_employees = random.randint(8, 12)
        
        for i in range(num_employees):
            employee_count += 1
            first_name = random.choice(first_names_pool)
            last_name = random.choice(last_names_pool)
            email = f"{first_name.lower()}.{last_name.lower()}{employee_count}@riskhub.test"
            name = f"{first_name} {last_name}"
            
            # First employee in Operations is the demo account
            if dept_name == "Operations" and i == 0:
                email = "ops.employee@riskhub.test"  # DEMO ACCOUNT
                name = "Operations Employee"
            
            existing = db.query(User).filter(User.email == email).first()
            if not existing:
                employee = User(
                    email=email,
                    name=name,
                    role_id=roles["employee"].id,
                    department_id=dept.id,
                    manager_id=dept_head.id if dept_head else None,
                    hashed_password=get_password_hash("test123")
                )
                db.add(employee)
    
    db.commit()
    
    # Set department heads' manager to CEO
    ceo = db_users.get("ceo@riskhub.test")
    if ceo:
        dept_heads_in_db = db.query(User).filter(
            User.role_id == roles["department_head"].id
        ).all()
        for head in dept_heads_in_db:
            head.manager_id = ceo.id
        
        # COO also reports to CEO
        coo = db_users.get("coo@riskhub.test")
        if coo:
            coo.manager_id = ceo.id
    
    db.commit()
    db.close()
    
    print(f"✅ Created {len(users_to_create) + employee_count} users")
    print(f"   - 4 C-Suite")
    print(f"   - 6 Governance roles")
    print(f"   - 10 Department heads")
    print(f"   - {employee_count} Employees")
    print(f"\n🔑 Demo Accounts:")
    print(f"   CRO: cro@riskhub.test / test123")
    print(f"   COO: coo@riskhub.test / test123")
    print(f"   Employee: ops.employee@riskhub.test / test123")

if __name__ == "__main__":
    seed_users()
```

def test_user_count(db_session):

    assert employee is not None
    assert employee.manager_id == coo.id
