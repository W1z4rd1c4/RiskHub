### Task 1: Create Resolved Approvals Seeding Script

#### [NEW] `backend/scripts/seed_e2e_resolved_approvals.py`

```python
"""
Phase 179-08: Resolved Approval Data Seeding
Seeds approval requests in terminal states for E2E tests.
"""
import asyncio
from datetime import datetime, timedelta
from sqlalchemy import select
from app.db.session import async_session_maker
from app.models import ApprovalRequest, Risk, User
from scripts.e2e_mappings import load_mappings
from app.schemas.approval import ApprovalStatus

# Resolved approval scenarios
RESOLVED_APPROVALS = [
    {
        "id_prefix": "E2E-APPROVED-RISK",
        "status": ApprovalStatus.APPROVED,
        "action": "DELETE",
        "notes": "Approved by Risk Manager for E2E test"
    },
    {
        "id_prefix": "E2E-REJECTED-RISK",
        "status": ApprovalStatus.REJECTED,
        "action": "DELETE",
        "notes": "Rejected - does not meet criteria"
    },
    {
        "id_prefix": "E2E-CANCELLED-RISK",
        "status": ApprovalStatus.CANCELLED,
        "action": "EDIT",
        "notes": "Cancelled by requester"
    },
    {
        "id_prefix": "E2E-TIERED-APPROVED",
        "status": ApprovalStatus.APPROVED,
        "action": "DELETE",
        "notes": "Tiered approval - privileged approver required",
        "requires_privileged": True
    },
]


async def seed_resolved_approvals():
    """Seed resolved approval requests for E2E tests."""
    async with async_session_maker() as db:
        users, depts = await load_mappings(db)
        
        # Check for existing resolved E2E approvals
        result = await db.execute(
            select(ApprovalRequest).where(
                ApprovalRequest.notes.contains("E2E-APPROVED") |
                ApprovalRequest.notes.contains("E2E-REJECTED") |
                ApprovalRequest.notes.contains("E2E-CANCELLED")
            )
        )
        existing = result.scalars().all()
        if existing:
            print(f"   ⏭️  Resolved approvals already seeded ({len(existing)} entries)")
            return
        
        # Get a sample risk for reference
        risk_result = await db.execute(
            select(Risk).where(Risk.name.contains("E2E")).limit(1)
        )
        sample_risk = risk_result.scalar_one_or_none()
        if not sample_risk:
            # Fallback to any risk
            risk_result = await db.execute(select(Risk).limit(1))
            sample_risk = risk_result.scalar_one_or_none()
        
        # User mappings
        requester_id = users.get("ops.head@riskhub.local")
        approver_id = users.get("risk.manager@riskhub.local")
        privileged_approver_id = users.get("cro@riskhub.local")
        
        created = 0
        base_time = datetime.utcnow() - timedelta(days=7)
        
        for i, scenario in enumerate(RESOLVED_APPROVALS):
            approval = ApprovalRequest(
                entity_type="RISK",
                entity_id=sample_risk.id if sample_risk else 1,
                action_type=scenario["action"],
                status=scenario["status"],
                requested_by_id=requester_id,
                requested_at=base_time + timedelta(days=i),
                resolved_by_id=privileged_approver_id if scenario.get("requires_privileged") else approver_id,
                resolved_at=base_time + timedelta(days=i, hours=2),
                primary_approver_id=approver_id,
                privileged_approver_id=privileged_approver_id if scenario.get("requires_privileged") else None,
                requires_privileged=scenario.get("requires_privileged", False),
                notes=scenario["notes"],
                proposed_changes={
                    "seeded_for": "e2e_tests",
                    "scenario": scenario["id_prefix"]
                },
            )
            db.add(approval)
            created += 1
        
        await db.commit()
        print(f"   ✓ Created {created} resolved approval requests")


if __name__ == "__main__":
    asyncio.run(seed_resolved_approvals())
```
