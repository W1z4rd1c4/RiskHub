## Context

### Why This Plan?

The `sensitive-fields/*.spec.ts` tests verify BUSINESS_LOGIC.md §6 rules:
- owner_id changes require approval
- department_id changes require approval
- category changes on risks require approval
- is_priority downgrade requires approval

These tests need existing pending approvals with specific `proposed_changes` JSON.

### Task 1: Create Sensitive Field Approvals Seeding Script

#### [NEW] `backend/scripts/seed_e2e_sensitive_approvals.py`

```python
"""
Phase 179-09: Sensitive Field Approval Data Seeding
Seeds pending approvals for sensitive field changes.
"""
import asyncio
from datetime import datetime
from sqlalchemy import select
from app.db.session import async_session_maker
from app.models import ApprovalRequest, Risk, Control, User
from scripts.e2e_mappings import load_mappings
from app.schemas.approval import ApprovalStatus

# Sensitive field change scenarios per BUSINESS_LOGIC.md §6
SENSITIVE_FIELD_SCENARIOS = [
    # Risk owner_id changes (§6.1)
    {
        "entity_type": "RISK",
        "action": "EDIT",
        "field": "owner_id",
        "description": "Change risk owner from Ops Head to Finance Head",
    },
    # Risk department_id changes (§6.1)
    {
        "entity_type": "RISK",
        "action": "EDIT",
        "field": "department_id",
        "description": "Move risk from Operations to Finance department",
    },
    # Risk category changes (§6.1)
    {
        "entity_type": "RISK",
        "action": "EDIT",
        "field": "category",
        "description": "Change risk category from Operational to Strategic",
    },
    # Risk is_priority downgrade (§6.3)
    {
        "entity_type": "RISK",
        "action": "EDIT",
        "field": "is_priority",
        "description": "Downgrade priority risk to non-priority",
    },
    # Control owner_id changes (§6.1)
    {
        "entity_type": "CONTROL",
        "action": "EDIT",
        "field": "control_owner_id",
        "description": "Change control owner to different department",
    },
    # Control department_id changes (§6.1)
    {
        "entity_type": "CONTROL",
        "action": "EDIT",
        "field": "department_id",
        "description": "Move control from IT to Operations department",
    },
    # NULL clearing scenario (§6.3)
    {
        "entity_type": "RISK",
        "action": "EDIT",
        "field": "owner_id",
        "description": "Clear owner (set to NULL)",
        "new_value": None,
    },
]


async def seed_sensitive_approvals():
    """Seed pending approvals for sensitive field changes."""
    async with async_session_maker() as db:
        users, depts = await load_mappings(db)
        
        # Check for existing sensitive field approvals
        result = await db.execute(
            select(ApprovalRequest).where(
                ApprovalRequest.notes.contains("E2E-SENSITIVE")
            )
        )
        existing = result.scalars().all()
        if existing:
            print(f"   ⏭️  Sensitive field approvals already seeded ({len(existing)} entries)")
            return
        
        # Get sample entities
        risk_result = await db.execute(select(Risk).limit(3))
        risks = risk_result.scalars().all()
        
        control_result = await db.execute(select(Control).limit(2))
        controls = control_result.scalars().all()
        
        # User mappings
        requester_id = users.get("ops.head@riskhub.local")
        new_owner_id = users.get("fin.head@riskhub.local")
        approver_id = users.get("risk.manager@riskhub.local")
        
        created = 0
        risk_index = 0
        control_index = 0
        
        for scenario in SENSITIVE_FIELD_SCENARIOS:
            entity_type = scenario["entity_type"]
            
            # Get entity reference
            if entity_type == "RISK" and risks:
                entity = risks[risk_index % len(risks)]
                entity_id = entity.id
                risk_index += 1
            elif entity_type == "CONTROL" and controls:
                entity = controls[control_index % len(controls)]
                entity_id = entity.id
                control_index += 1
            else:
                continue
            
            # Build proposed changes
            new_value = scenario.get("new_value", new_owner_id if "owner" in scenario["field"] else depts.get("Finance"))
            proposed_changes = {
                "field": scenario["field"],
                "old_value": "original_value",
                "new_value": new_value,
                "seeded_for": "e2e_sensitive_tests",
            }
            
            approval = ApprovalRequest(
                entity_type=entity_type,
                entity_id=entity_id,
                action_type=scenario["action"],
                status=ApprovalStatus.PENDING,
                requested_by_id=requester_id,
                requested_at=datetime.utcnow(),
                primary_approver_id=approver_id,
                notes=f"E2E-SENSITIVE: {scenario['description']}",
                proposed_changes=proposed_changes,
            )
            db.add(approval)
            created += 1
        
        await db.commit()
        print(f"   ✓ Created {created} sensitive field approval requests")


if __name__ == "__main__":
    asyncio.run(seed_sensitive_approvals())
```
