## Tasks

### Task 1: Create Control Seeding Script

#### [NEW] `backend/scripts/seed_e2e_controls.py`

```python
"""
Phase 179-03: Cross-Department Control Data
Creates 12 controls with risk linkages for E2E testing.
"""
import asyncio
from sqlalchemy import select
from app.db.session import async_session_maker
from app.models import Control, Risk, ControlRiskLink, ControlFrequency, ControlForm, ControlStatus
from scripts.e2e_mappings import load_mappings

CONTROLS = [
    {
        "name": "E2E-CTRL-001 Motor UW Pricing Validation",
        "description": "Automated validation of motor insurance pricing against actuarial models",
        "dept": "Operations",
        "owner": "ops.analyst@riskhub.local",
        "frequency": "daily",
        "control_form": "automatic",
        "risk_links": ["E2E-UW-001"],
    },
    # ... (11 more controls)
]

async def seed_controls():
    """Create E2E test controls with risk linkages."""
    async with async_session_maker() as db:
        users, depts = await load_mappings(db)
        
        created_controls = 0
        created_links = 0
        
        for ctrl_data in CONTROLS:
            # Check if exists
            result = await db.execute(
                select(Control).where(Control.name == ctrl_data["name"])
            )
            if result.scalar_one_or_none():
                continue
            
            # Resolve IDs
            owner_id = users[ctrl_data.pop("owner")]
            dept_id = depts[ctrl_data.pop("dept")]
            risk_codes = ctrl_data.pop("risk_links")
            
            control = Control(
                name=ctrl_data["name"],
                description=ctrl_data["description"],
                department_id=dept_id,
                control_owner_id=owner_id,
                frequency=ctrl_data.get("frequency", "monthly"),
                control_form=ctrl_data.get("control_form", "manual"),
                status=ControlStatus.active.value,
                risk_level=3,
            )
            db.add(control)
            await db.flush()
            created_controls += 1
            
            # Create risk linkages
            for risk_code in risk_codes:
                result = await db.execute(
                    select(Risk).where(Risk.risk_id_code == risk_code)
                )
                risk = result.scalar_one_or_none()
                if risk:
                    link = ControlRiskLink(
                        control_id=control.id,
                        risk_id=risk.id,
                        effectiveness="high",
                    )
                    db.add(link)
                    created_links += 1
        
        await db.commit()
        print(f"✅ Created {created_controls} controls, {created_links} risk links")

if __name__ == "__main__":
    asyncio.run(seed_controls())
```

### Step 3: Verify High-Risk Links
```bash
cd backend  
python -c "
import asyncio
from sqlalchemy import select
from app.db.session import async_session_maker
from app.models import Control, Risk, ControlRiskLink

async def check():
    async with async_session_maker() as db:
        result = await db.execute(
            select(Control, Risk)
            .join(ControlRiskLink, Control.id == ControlRiskLink.control_id)
            .join(Risk, ControlRiskLink.risk_id == Risk.id)
            .where(Control.name.like('E2E-%'))
            .where((Risk.is_priority == True) | (Risk.net_score >= 10))
        )
        high_risk_links = 0
        for ctrl, risk in result:
            high_risk_links += 1
            print(f'{ctrl.name} → {risk.risk_id_code} (priority={risk.is_priority}, net={risk.net_score})')
        print(f'\\nHigh-risk linked controls: {high_risk_links}/7')

asyncio.run(check())
"
```
