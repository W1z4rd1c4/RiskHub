Check counts:
```bash
cd backend && python -c "
from sqlalchemy import select, func
from app.db.session import async_session_maker
from app.models import ApprovalRequest, ApprovalStatus
import asyncio

async def check():
    async with async_session_maker() as db:
        pending = (await db.execute(select(func.count()).select_from(ApprovalRequest).where(ApprovalRequest.status == ApprovalStatus.pending))).scalar()
        print(f'PENDING: {pending}')
asyncio.run(check())
"
```
