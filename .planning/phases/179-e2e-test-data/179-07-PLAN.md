# Plan 179-07: Activity Log Data Seeding

<objective>
Seed activity log entries for all entity types (RISK, CONTROL, KRI, APPROVAL) with CREATE, UPDATE, and ARCHIVE actions. This enables activity-logging E2E tests that currently skip due to missing historical data.
</objective>

## Context

### Why This Plan?

The `activity-logging/*.spec.ts` tests skip because they look for specific activity log entry types:
- RISK: CREATE, UPDATE, ARCHIVE
- CONTROL: CREATE, UPDATE, ARCHIVE
- KRI: CREATE, UPDATE
- KRI_VALUE: CREATE (value submission)
- APPROVAL: CREATE, APPROVE, REJECT, CANCEL

### Current Skip Count
~20 tests across 3 spec files in `e2e/activity-logging/`

## Tasks

### Task 1: Create Activity Log Seeding Script

#### [NEW] `backend/scripts/seed_e2e_activity_logs.py`

```python
"""
Phase 179-07: Activity Log Data Seeding
Seeds activity log entries for E2E tests.
"""
import asyncio
from datetime import datetime, timedelta
from sqlalchemy import select
from app.db.session import async_session_maker
from app.models import ActivityLog, Risk, Control, KRI
from scripts.e2e_mappings import load_mappings

# Activity types we need to seed
ENTITY_ACTIONS = [
    ("RISK", "CREATE"),
    ("RISK", "UPDATE"),
    ("RISK", "ARCHIVE"),
    ("CONTROL", "CREATE"),
    ("CONTROL", "UPDATE"),
    ("CONTROL", "ARCHIVE"),
    ("KRI", "CREATE"),
    ("KRI", "UPDATE"),
    ("KRI_VALUE", "CREATE"),
    ("APPROVAL", "CREATE"),
    ("APPROVAL", "APPROVE"),
    ("APPROVAL", "REJECT"),
    ("APPROVAL", "CANCEL"),
]


async def seed_activity_logs():
    """Seed activity log entries for E2E tests."""
    async with async_session_maker() as db:
        users, depts = await load_mappings(db)
        
        # Check for existing E2E activity logs
        result = await db.execute(
            select(ActivityLog).where(ActivityLog.notes.contains("E2E-SEED"))
        )
        existing = result.scalars().all()
        if existing:
            print(f"   ⏭️  Activity logs already seeded ({len(existing)} entries)")
            return
        
        # Get a sample entity for each type
        risk_result = await db.execute(select(Risk).limit(1))
        sample_risk = risk_result.scalar_one_or_none()
        
        control_result = await db.execute(select(Control).limit(1))
        sample_control = control_result.scalar_one_or_none()
        
        kri_result = await db.execute(select(KRI).limit(1))
        sample_kri = kri_result.scalar_one_or_none()
        
        cro_id = users.get("cro@riskhub.local")
        rm_id = users.get("risk.manager@riskhub.local")
        
        created = 0
        base_time = datetime.utcnow() - timedelta(hours=24)
        
        for i, (entity_type, action) in enumerate(ENTITY_ACTIONS):
            # Determine resource_id based on entity type
            resource_id = None
            if entity_type == "RISK" and sample_risk:
                resource_id = sample_risk.id
            elif entity_type == "CONTROL" and sample_control:
                resource_id = sample_control.id
            elif entity_type in ("KRI", "KRI_VALUE") and sample_kri:
                resource_id = sample_kri.id
            
            # Vary the user for diversity
            user_id = cro_id if i % 2 == 0 else rm_id
            
            entry = ActivityLog(
                entity_type=entity_type,
                action_type=action,
                resource_id=resource_id,
                resource_name=f"E2E-SEED-{entity_type}-{action}",
                user_id=user_id,
                department_id=depts.get("Risk Management"),
                notes=f"E2E-SEED: Seeded for activity logging tests",
                changes={"seeded_for": "e2e_tests", "action": action},
                created_at=base_time + timedelta(minutes=i*5),
            )
            db.add(entry)
            created += 1
        
        await db.commit()
        print(f"   ✓ Created {created} activity log entries")


if __name__ == "__main__":
    asyncio.run(seed_activity_logs())
```
