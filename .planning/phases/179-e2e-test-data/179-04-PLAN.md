## Tasks

### Task 1: Create KRI Seeding Script

#### [NEW] `backend/scripts/seed_e2e_kris.py`

```python
"""
Phase 179-04: KRI Data with Reporting Owners
Creates 10 KRIs linked to E2E risks for testing.
"""
import asyncio
from sqlalchemy import select
from app.db.session import async_session_maker
from app.models import KeyRiskIndicator, Risk
from scripts.e2e_mappings import load_mappings

KRIS = [
    {
        "name": "E2E-KRI-001 Motor Loss Ratio",
        "description": "Ratio of claims paid to premiums earned for motor insurance",
        "risk_code": "E2E-UW-001",
        "reporting_owner": "fin.analyst@riskhub.local",
        "unit": "%",
        "lower_limit": 0,
        "upper_limit": 65,
        "current_value": 58,
        "frequency": "monthly",
    },
    {
        "name": "E2E-KRI-002 Travel Claim Frequency",
        "description": "Daily count of travel insurance claims submitted",
        "risk_code": "E2E-UW-002",
        "reporting_owner": "ops.analyst@riskhub.local",
        "unit": "count/day",
        "lower_limit": 0,
        "upper_limit": 5,
        "current_value": 3,
        "frequency": "daily",
    },
    # ... (8 more KRIs following the pattern)
]

async def seed_kris():
    """Create E2E test KRIs."""
    async with async_session_maker() as db:
        users, _ = await load_mappings(db)
        
        created = 0
        skipped = 0
        
        for kri_data in KRIS:
            # Check if exists
            result = await db.execute(
                select(KeyRiskIndicator).where(KeyRiskIndicator.name == kri_data["name"])
            )
            if result.scalar_one_or_none():
                skipped += 1
                continue
            
            # Get linked risk
            result = await db.execute(
                select(Risk).where(Risk.risk_id_code == kri_data.pop("risk_code"))
            )
            risk = result.scalar_one_or_none()
            if not risk:
                print(f"⚠️ Risk not found for {kri_data['name']}")
                continue
            
            # Resolve reporting owner
            reporting_owner_id = users.get(kri_data.pop("reporting_owner"))
            
            kri = KeyRiskIndicator(
                name=kri_data["name"],
                description=kri_data["description"],
                risk_id=risk.id,
                reporting_owner_id=reporting_owner_id,
                unit=kri_data["unit"],
                lower_limit=kri_data["lower_limit"],
                upper_limit=kri_data["upper_limit"],
                current_value=kri_data["current_value"],
                frequency=kri_data.get("frequency", "monthly"),
            )
            db.add(kri)
            created += 1
        
        await db.commit()
        print(f"✅ Created {created} KRIs, skipped {skipped} existing")

if __name__ == "__main__":
    asyncio.run(seed_kris())
```

### Step 3: Verify Cross-Department Reporting Owners
```bash
cd backend
python -c "
import asyncio
from sqlalchemy import select
from app.db.session import async_session_maker
from app.models import KeyRiskIndicator, Risk, User

async def check():
    async with async_session_maker() as db:
        result = await db.execute(
            select(KeyRiskIndicator, Risk, User)
            .join(Risk, KeyRiskIndicator.risk_id == Risk.id)
            .outerjoin(User, KeyRiskIndicator.reporting_owner_id == User.id)
            .where(KeyRiskIndicator.name.like('E2E-%'))
        )
        cross_dept = 0
        for kri, risk, owner in result:
            if owner and owner.department_id != risk.department_id:
                cross_dept += 1
                print(f'{kri.name}: Risk in dept {risk.department_id}, owner in dept {owner.department_id}')
        print(f'\\nCross-department reporting owners: {cross_dept}/4')

asyncio.run(check())
"
```
