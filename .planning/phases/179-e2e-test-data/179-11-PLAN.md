# Plan 179-11: Deterministic Cross-Department Scenarios

<objective>
Create deterministic cross-department ownership scenarios with known user-entity relationships for predictable E2E access tests.
</objective>

---

## Context

### Why This Plan?

The `cross-department/*.spec.ts` tests often skip because:
1. Tests rely on finding entities owned by specific cross-department users
2. Current E2E data has semi-random ownership assignments
3. Tests cannot predict which entities a user will own in another department

We need **deterministic** scenarios where:
- User A in Department X explicitly owns Entity Y in Department Z
- Tests can rely on these known relationships

### Current Skip Count
~44 tests across 4 spec files in `e2e/cross-department/`

---

## Tasks

### Task 1: Create Deterministic Cross-Department Seeding Script

#### [NEW] `backend/scripts/seed_e2e_cross_dept.py`

```python
"""
Phase 179-11: Deterministic Cross-Department Scenarios
Creates known user-entity ownership relationships across departments.
"""
import asyncio
from datetime import datetime
from sqlalchemy import select
from app.db.session import async_session_maker
from app.models import Risk, Control, KRI, User, Department
from scripts.e2e_mappings import load_mappings

# Deterministic cross-department scenarios
# Key: test scenario name
# Value: ownership configuration
CROSS_DEPT_SCENARIOS = {
    # Finance user owns Operations risk
    "fin_owns_ops_risk": {
        "owner_email": "fin.head@riskhub.local",
        "entity_type": "RISK",
        "entity_dept": "Operations",
        "name_prefix": "E2E-XDEPT-FIN-OPS-RISK",
    },
    # IT user owns Finance risk
    "it_owns_fin_risk": {
        "owner_email": "it.head@riskhub.local",
        "entity_type": "RISK",
        "entity_dept": "Finance",
        "name_prefix": "E2E-XDEPT-IT-FIN-RISK",
    },
    # Operations analyst owns IT control
    "ops_owns_it_control": {
        "owner_email": "ops.analyst@riskhub.local",
        "entity_type": "CONTROL",
        "entity_dept": "IT",
        "name_prefix": "E2E-XDEPT-OPS-IT-CTRL",
    },
    # Finance analyst reports on IT KRI
    "fin_reports_it_kri": {
        "owner_email": "fin.analyst@riskhub.local",
        "entity_type": "KRI",
        "entity_dept": "IT",
        "name_prefix": "E2E-XDEPT-FIN-IT-KRI",
    },
    # IT analyst owns Operations control
    "it_owns_ops_control": {
        "owner_email": "it.analyst@riskhub.local",
        "entity_type": "CONTROL",
        "entity_dept": "Operations",
        "name_prefix": "E2E-XDEPT-IT-OPS-CTRL",
    },
}


async def seed_cross_dept_scenarios():
    """Seed deterministic cross-department ownership scenarios."""
    async with async_session_maker() as db:
        users, depts = await load_mappings(db)
        
        # Check if already seeded
        result = await db.execute(
            select(Risk).where(Risk.name.contains("E2E-XDEPT"))
        )
        if result.scalars().first():
            print("   ⏭️  Cross-department scenarios already seeded")
            return
        
        created = 0
        
        for scenario_name, config in CROSS_DEPT_SCENARIOS.items():
            owner_id = users.get(config["owner_email"])
            dept_id = depts.get(config["entity_dept"])
            
            if not owner_id or not dept_id:
                print(f"   ⚠️  Skipping {scenario_name}: missing user/dept")
                continue
            
            entity_type = config["entity_type"]
            name = f"{config['name_prefix']}-{scenario_name.upper()}"
            
            if entity_type == "RISK":
                # Need a linked risk for KRI tests
                existing = await db.execute(
                    select(Risk).where(Risk.name == name)
                )
                if not existing.scalar_one_or_none():
                    entity = Risk(
                        name=name,
                        risk_id=f"XDEPT-{created+1:03d}",
                        description=f"Cross-department test: {scenario_name}",
                        department_id=dept_id,
                        owner_id=owner_id,
                        category="Operational",
                        gross_likelihood=3,
                        gross_impact=3,
                        net_likelihood=2,
                        net_impact=2,
                        created_by_id=users.get("cro@riskhub.local"),
                        archived=False,
                    )
                    db.add(entity)
                    created += 1
            
            elif entity_type == "CONTROL":
                existing = await db.execute(
                    select(Control).where(Control.name == name)
                )
                if not existing.scalar_one_or_none():
                    entity = Control(
                        name=name,
                        control_id=f"XDEPT-C-{created+1:03d}",
                        description=f"Cross-department test: {scenario_name}",
                        department_id=dept_id,
                        control_owner_id=owner_id,
                        control_type="Detective",
                        created_by_id=users.get("cro@riskhub.local"),
                        archived=False,
                    )
                    db.add(entity)
                    created += 1
            
            elif entity_type == "KRI":
                # KRI needs a linked risk - use existing E2E risk
                risk_result = await db.execute(
                    select(Risk).where(Risk.name.contains("E2E")).limit(1)
                )
                linked_risk = risk_result.scalar_one_or_none()
                
                if linked_risk:
                    existing = await db.execute(
                        select(KRI).where(KRI.name == name)
                    )
                    if not existing.scalar_one_or_none():
                        entity = KRI(
                            name=name,
                            description=f"Cross-department test: {scenario_name}",
                            risk_id=linked_risk.id,
                            reporting_owner_id=owner_id,
                            unit="Percent",
                            frequency="Monthly",
                            target_value=50.0,
                            warning_threshold=70.0,
                            critical_threshold=90.0,
                            created_by_id=users.get("cro@riskhub.local"),
                            archived=False,
                        )
                        db.add(entity)
                        created += 1
        
        await db.commit()
        print(f"   ✓ Created {created} cross-department entities")


if __name__ == "__main__":
    asyncio.run(seed_cross_dept_scenarios())
```
