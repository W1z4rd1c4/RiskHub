### Task 1: Create Permission-Gated Actions Seeding Script

#### [NEW] `backend/scripts/seed_e2e_permission_actions.py`

```python
"""
Phase 179-10: Permission-Gated Action Data Seeding
Seeds delete approvals, control executions, and KRI corrections.
"""
import asyncio
from datetime import datetime, timedelta
from sqlalchemy import select
from app.db.session import async_session_maker
from app.models import (
    ApprovalRequest, Risk, Control, KRI, KRIValueHistory,
    ControlExecution, ActivityLog
)
from scripts.e2e_mappings import load_mappings
from app.schemas.approval import ApprovalStatus


async def seed_delete_approvals(db, users, depts, risks, controls):
    """Seed DELETE approval requests from non-privileged users."""
    # Check if already seeded
    result = await db.execute(
        select(ApprovalRequest).where(
            ApprovalRequest.notes.contains("E2E-DELETE-PERMISSION")
        )
    )
    if result.scalars().first():
        print("   ⏭️  Delete approvals already seeded")
        return 0
    
    requester_id = users.get("ops.analyst@riskhub.local")  # Non-privileged
    approver_id = users.get("ops.head@riskhub.local")  # Dept Head
    
    created = 0
    
    # Risk delete approval
    if risks:
        approval = ApprovalRequest(
            entity_type="RISK",
            entity_id=risks[0].id,
            action_type="DELETE",
            status=ApprovalStatus.PENDING,
            requested_by_id=requester_id,
            requested_at=datetime.utcnow() - timedelta(hours=12),
            primary_approver_id=approver_id,
            notes="E2E-DELETE-PERMISSION: Employee requests risk deletion",
        )
        db.add(approval)
        created += 1
    
    # Control delete approval
    if controls:
        approval = ApprovalRequest(
            entity_type="CONTROL",
            entity_id=controls[0].id,
            action_type="DELETE",
            status=ApprovalStatus.PENDING,
            requested_by_id=requester_id,
            requested_at=datetime.utcnow() - timedelta(hours=6),
            primary_approver_id=approver_id,
            notes="E2E-DELETE-PERMISSION: Employee requests control deletion",
        )
        db.add(approval)
        created += 1
    
    return created


async def seed_control_executions(db, users, controls):
    """Seed control execution log entries."""
    # Check if already seeded
    result = await db.execute(
        select(ControlExecution).where(
            ControlExecution.notes.contains("E2E-EXECUTION")
        )
    )
    if result.scalars().first():
        print("   ⏭️  Control executions already seeded")
        return 0
    
    executor_id = users.get("ops.analyst@riskhub.local")
    created = 0
    
    for i, control in enumerate(controls[:3]):  # First 3 controls
        execution = ControlExecution(
            control_id=control.id,
            executed_by_id=executor_id,
            executed_at=datetime.utcnow() - timedelta(days=i, hours=i*2),
            outcome="EFFECTIVE" if i % 2 == 0 else "PARTIALLY_EFFECTIVE",
            notes=f"E2E-EXECUTION: Quarterly control test #{i+1}",
        )
        db.add(execution)
        created += 1
    
    return created


async def seed_kri_corrections(db, users, kris):
    """Seed KRI value correction entries."""
    # Check for existing corrections
    result = await db.execute(
        select(KRIValueHistory).where(
            KRIValueHistory.notes.contains("E2E-CORRECTION")
        )
    )
    if result.scalars().first():
        print("   ⏭️  KRI corrections already seeded")
        return 0
    
    corrector_id = users.get("risk.manager@riskhub.local")
    created = 0
    
    for i, kri in enumerate(kris[:2]):  # First 2 KRIs
        # Create original value
        original = KRIValueHistory(
            kri_id=kri.id,
            value=100.0 + i * 10,
            recorded_by_id=users.get("fin.analyst@riskhub.local"),
            recorded_at=datetime.utcnow() - timedelta(days=3),
            notes="Original submission",
            is_correction=False,
        )
        db.add(original)
        
        # Create correction
        correction = KRIValueHistory(
            kri_id=kri.id,
            value=105.0 + i * 10,  # Corrected value
            recorded_by_id=corrector_id,
            recorded_at=datetime.utcnow() - timedelta(days=1),
            notes=f"E2E-CORRECTION: Corrected calculation error",
            is_correction=True,
            corrects_id=None,  # Will be set after commit if needed
        )
        db.add(correction)
        created += 2
    
    return created


async def seed_permission_actions():
    """Main entry point."""
    async with async_session_maker() as db:
        users, depts = await load_mappings(db)
        
        # Get sample entities
        risk_result = await db.execute(select(Risk).limit(3))
        risks = risk_result.scalars().all()
        
        control_result = await db.execute(select(Control).limit(5))
        controls = control_result.scalars().all()
        
        kri_result = await db.execute(select(KRI).limit(3))
        kris = kri_result.scalars().all()
        
        total = 0
        
        # Seed delete approvals
        count = await seed_delete_approvals(db, users, depts, risks, controls)
        total += count
        
        # Seed control executions
        count = await seed_control_executions(db, users, controls)
        total += count
        
        # Seed KRI corrections
        count = await seed_kri_corrections(db, users, kris)
        total += count
        
        await db.commit()
        print(f"   ✓ Created {total} permission-gated action entries")


if __name__ == "__main__":
    asyncio.run(seed_permission_actions())
```
