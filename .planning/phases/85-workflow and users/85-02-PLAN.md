<task type="checkpoint:decision" gate="blocking">
  <decision>Choose how "privileged account" is modeled</decision>
  <context>
    Admin/CRO must be able to designate privileged accounts; this choice affects scope logic,
    API fields, and UI behavior. Keep existing behavior for admin/cro/risk_manager equivalent after migration.
  </context>
  <options>
    <option id="user-flag">
      <name>User-level privileged flag (User.is_privileged)</name>
      <pros>Per-user control, matches "who is privileged" requirement, minimal role changes</pros>
      <cons>Privileges can diverge from role definitions; needs guardrails to avoid lockout</cons>
    </option>
    <option id="role-flag">
      <name>Role-level privileged flag (Role.is_privileged)</name>
      <pros>Cleaner RBAC model, fewer overrides, easy to reason about</pros>
      <cons>Cannot make a single user privileged without role change; affects all users in role</cons>
    </option>
    <option id="access-scope">
      <name>User access_scope enum (global/department/manager)</name>
      <pros>Separates permissions from data scope; enterprise-friendly</pros>
      <cons>More refactor work; changes scoping rules across endpoints</cons>
    </option>
  </options>
  <resume-signal>Select: user-flag, role-flag, or access-scope</resume-signal>
</task>
