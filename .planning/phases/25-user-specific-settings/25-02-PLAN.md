Refactor ThemeContext to use server-synced storage:

1. Import storage utilities:
   ```typescript
   import { getLocalTheme, saveThemeToServer } from '@/utils/userSettingsStorage';
   import { useAuth } from '@/contexts/AuthContext';
   ```

2. Add user awareness:
   ```typescript
   const { isAuthenticated } = useAuth();
   ```

3. Update initial state to use local storage:
   ```typescript
   const [theme, setThemeState] = useState<Theme>(() => {
     if (typeof window === 'undefined') return 'riskhub';
     const stored = getLocalTheme();
     if (stored === 'light' || stored === 'dark' || stored === 'riskhub') {
       return stored;
     }
     return 'riskhub';
   });
   ```

4. Update setTheme to sync to server when authenticated:
   ```typescript
   const setTheme = (newTheme: Theme) => {
     setThemeState(newTheme);
     if (isAuthenticated) {
       saveThemeToServer(newTheme).catch(console.error);
     } else {
       // Guest mode: just save locally
       localStorage.setItem('riskhub-theme', newTheme);
     }
   };
   ```

5. Listen for storage changes (for multi-tab sync):
   ```typescript
   useEffect(() => {
     const handleStorageChange = (e: StorageEvent) => {
       if (e.key === 'riskhub-theme' && e.newValue) {
         const newTheme = e.newValue as Theme;
         if (newTheme === 'light' || newTheme === 'dark' || newTheme === 'riskhub') {
           setThemeState(newTheme);
         }
       }
     };
     window.addEventListener('storage', handleStorageChange);
     return () => window.removeEventListener('storage', handleStorageChange);
   }, []);
   ```
  </action>
