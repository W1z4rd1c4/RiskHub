2. Add user awareness and sync on change:
   ```typescript
   export function useLanguage() {
     const { i18n: i18nInstance } = useI18nextTranslation();
     const { isAuthenticated } = useAuth();
     
     const language = i18nInstance.language as SupportedLanguage;
     
     const setLanguage = useCallback(
       (newLang: SupportedLanguage) => {
         i18nInstance.changeLanguage(newLang);
         if (isAuthenticated) {
           saveLanguageToServer(newLang).catch(console.error);
         } else {
           localStorage.setItem(STORAGE_KEY, newLang);
         }
       },
       [i18nInstance, isAuthenticated]
     );
     
     return useMemo(
       () => ({ language, setLanguage }),
       [language, setLanguage]
     );
   }
   ```

   - On login, syncPreferencesFromServer() updates localStorage
   - This ensures new logins get correct language

The key flow:

- Login: AuthContext calls syncPreferencesFromServer() → updates localStorage → triggers re-render
- Language change: Update i18n + local + server (async)

Enhance syncPreferencesFromServer to trigger i18n language change:

```typescript
import i18n from '@/i18n';

export async function syncPreferencesFromServer(): Promise<UserPreferences> {
  const prefs = await preferencesApi.get();
  
  // Update local cache with server values
  if (prefs.theme) setLocalTheme(prefs.theme);
  if (prefs.language) {
    setLocalLanguage(prefs.language);
    // Also update i18n instance
    if (i18n.language !== prefs.language) {
      i18n.changeLanguage(prefs.language);
    }
  }
  
  return prefs;
}
```

This ensures when a user logs in on a new device, their language preference is applied immediately.
