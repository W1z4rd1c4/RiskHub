### 3. Create Concurrency Stress Test

**File**: `backend/tests/test_concurrency_stress.py`

```python
"""Stress tests for concurrency fixes."""
import asyncio
import pytest
from httpx import AsyncClient


@pytest.mark.asyncio
async def test_50_concurrent_risk_creations(async_client, auth_headers, test_department):
    """50 simultaneous risk creations should all succeed with unique IDs."""
    async def create_risk(i):
        return await async_client.post(
            "/api/v1/risks",
            json={
                "name": f"Stress Risk {i}",
                "process": "STRESS",
                "risk_type": "operational",
                "description": f"Stress test {i}",
                "department_id": test_department["id"],
                "gross_probability": 3, "gross_impact": 3,
                "net_probability": 2, "net_impact": 2,
                "status": "active",
            },
            headers=auth_headers,
        )
    
    results = await asyncio.gather(*[create_risk(i) for i in range(50)])
    
    successes = [r for r in results if r.status_code == 201]
    assert len(successes) == 50, f"Only {len(successes)}/50 succeeded"
    
    ids = [r.json()["risk_id_code"] for r in successes]
    assert len(set(ids)) == 50, f"Duplicate IDs found"


@pytest.mark.asyncio
async def test_20_concurrent_approval_requests(async_client, auth_headers_dept_head, test_risk):
    """20 concurrent delete requests should result in only 1 pending approval."""
    async def delete_risk():
        return await async_client.delete(
            f"/api/v1/risks/{test_risk['id']}",
            params={"reason": "Concurrent test"},
            headers=auth_headers_dept_head,
        )
    
    results = await asyncio.gather(*[delete_risk() for _ in range(20)])
    
    successes = [r for r in results if r.status_code == 202]
    conflicts = [r for r in results if r.status_code == 409]
    
    assert len(successes) == 1, f"Expected 1 success, got {len(successes)}"
    assert len(conflicts) == 19, f"Expected 19 conflicts, got {len(conflicts)}"
```
