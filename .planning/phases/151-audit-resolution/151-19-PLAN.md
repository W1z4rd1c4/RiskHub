3. **Solution**: Add the commit inside the `else` block instead of having it unconditionally at line 520:

   ```python
       primary_approver_id = risk.owner_id if risk.owner_id != current_user.id else None
       
       # Fallback to department head if no owner or self-approval
       if not primary_approver_id and risk.department_id:
           from sqlalchemy import select as sa_select
           from app.models import Department
           dept_result = await db.execute(sa_select(Department).where(Department.id == risk.department_id))

           resource_id=risk.id,

           requested_by_id=current_user.id,
           reason=f"{reason}\n\nDescription: {desc_snippet}" if desc_snippet else reason,

           action_type=ApprovalActionType.DELETE,
           primary_approver_id=primary_approver_id,
           requires_privileged_approval=requires_privileged,
       )
   ```

```python
@pytest.mark.asyncio
async def test_cancel_pending_privileged_request(
    auth_client: AsyncClient,
    db_session: AsyncSession,
    test_risk,
    test_user,
):
    """Verify user can cancel their request even after it moves to PENDING_PRIVILEGED."""
    # Create an approval request already in PENDING_PRIVILEGED state
    approval = ApprovalRequest(
        resource_type=ApprovalResourceType.RISK,
        resource_id=test_risk.id,
        resource_name=test_risk.name,
        requested_by_id=test_user.id,  # Ensure same user as auth_client
        reason="Test cancel PENDING_PRIVILEGED",
        status=ApprovalStatus.PENDING_PRIVILEGED,
    )
    db_session.add(approval)
    await db_session.commit()
    await db_session.refresh(approval)
    
    # Cancel it
    response = await auth_client.post(f"/api/v1/approvals/{approval.id}/cancel")
    assert response.status_code == 200
    assert response.json()["status"] == "cancelled"
```

#### Test 2: Activity Log Created on Cancel
```python
@pytest.mark.asyncio
async def test_cancel_creates_activity_log(
    auth_client: AsyncClient,
    db_session: AsyncSession,
    test_risk,
):
    """Verify cancelling an approval creates an activity log entry."""
    from app.models.activity_log import ActivityLog, ActivityAction, ActivityEntityType
    
    # Create request
    create_response = await auth_client.post(
        "/api/v1/approvals",
        json={
            "resource_type": "risk",
            "resource_id": test_risk.id,
            "reason": "Test cancel logging"
        }
    )
    approval_id = create_response.json()["id"]
    
    # Cancel it
    await auth_client.post(f"/api/v1/approvals/{approval_id}/cancel")
    
    # Check activity log
    result = await db_session.execute(
        select(ActivityLog).where(
            ActivityLog.entity_type == ActivityEntityType.APPROVAL,
            ActivityLog.entity_id == approval_id,
            ActivityLog.action == ActivityAction.DELETE,
        )
    )
    log_entry = result.scalar_one_or_none()
    assert log_entry is not None
    assert "cancelled" in log_entry.description.lower()
```
