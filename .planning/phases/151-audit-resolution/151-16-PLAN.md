# Plan 151-16: Approval Request DB-level Constraints

## Objective
Enforce business rules for pending approvals at the database level to prevent duplicate requests for the same action on the same resource.

## Context & Problem Statement

Currently in [`backend/app/api/v1/endpoints/risks.py`](file:///Users/stefanlesnak/Antigravity/Risk%20App/backend/app/api/v1/endpoints/risks.py#L566-L575), the duplicate check happens only in application code:

```python
# Lines 566-575 in delete_risk:
existing = await db.execute(
    select(ApprovalRequest).where(
        ApprovalRequest.resource_type == ApprovalResourceType.RISK,
        ApprovalRequest.resource_id == risk.id,
        ApprovalRequest.status == ApprovalStatus.PENDING
    )
)
if existing.scalar_one_or_none():
    raise HTTPException(status_code=400, detail="Deletion request already pending")
```

**The Bug**: Two concurrent requests can:
1. Both execute the `SELECT` and find no existing pending request
2. Both proceed to `db.add(approval)`
3. Both commit successfully → Two pending approvals for same item!

## Execution Context
- **Model File**: `backend/app/models/approval_request.py`
- **Endpoint Files**: `backend/app/api/v1/endpoints/risks.py`, `backend/app/api/v1/endpoints/controls.py`
- **Migration**: New Alembic migration required
- **Related Test Files**: New `backend/tests/test_approval_constraints.py`

def upgrade() -> None:
    """Add partial unique index for pending approval requests."""
    # PostgreSQL syntax (production)
    # This prevents duplicate PENDING approvals for same (resource_type, resource_id, action_type)
    op.execute("""
        CREATE UNIQUE INDEX IF NOT EXISTS ux_approval_pending 
        ON approval_requests (resource_type, resource_id, action_type) 
        WHERE status = 'PENDING' OR status = 'PENDING_PRIVILEGED'
    """)


def downgrade() -> None:
    """Remove the partial unique index."""
    op.execute("DROP INDEX IF EXISTS ux_approval_pending")
```

"""Tests for approval request database constraints."""
import asyncio
import pytest
from httpx import AsyncClient
from sqlalchemy import select

from app.models import ApprovalRequest, ApprovalStatus


@pytest.mark.asyncio
async def test_concurrent_delete_requests_only_one_succeeds(
    async_client: AsyncClient,
    auth_headers_department_head: dict,
    test_risk: dict,
):
    """
    Verify that concurrent delete requests result in only one pending approval.
    
    This tests the partial unique index `ux_approval_pending`.
    """
    risk_id = test_risk["id"]
    
    # Fire 5 delete requests concurrently
    async def delete_risk():
        return await async_client.delete(
            f"/api/v1/risks/{risk_id}",
            params={"reason": "Concurrent delete test"},
            headers=auth_headers_department_head,  # Non-privileged user
        )
    
    results = await asyncio.gather(*[delete_risk() for _ in range(5)])
    
    # Count successes (202) and conflicts (409)
    successes = [r for r in results if r.status_code == 202]
    conflicts = [r for r in results if r.status_code == 409]
    
    # Exactly one should succeed, rest should be 409
    assert len(successes) == 1, f"Expected 1 success, got {len(successes)}"
    assert len(conflicts) == 4, f"Expected 4 conflicts, got {len(conflicts)}"
    
    # Verify conflict message
    for r in conflicts:
        assert "already pending" in r.json()["detail"].lower()


@pytest.mark.asyncio
async def test_different_actions_allowed_simultaneously(
    async_client: AsyncClient,
    auth_headers_department_head: dict,
    test_risk: dict,
    db_session,
):
    """
    Verify that DELETE and EDIT requests for same resource can both be pending.
    
    The constraint only prevents duplicates of the SAME action type.
    """
    risk_id = test_risk["id"]
    
    # Create delete request
    delete_response = await async_client.delete(
        f"/api/v1/risks/{risk_id}",
        params={"reason": "Delete request"},
        headers=auth_headers_department_head,
    )
    assert delete_response.status_code == 202
    
    # Create edit request (changing a sensitive field)
    edit_response = await async_client.patch(
        f"/api/v1/risks/{risk_id}",
        json={"is_priority": False},  # Triggers approval for priority downgrade
        headers=auth_headers_department_head,
    )
    # This should fail because we block updates while delete is pending
    # (business rule, not constraint)
    assert edit_response.status_code == 409
    assert "deletion is pending" in edit_response.json()["detail"]


@pytest.mark.asyncio
async def test_no_duplicate_edit_requests(
    async_client: AsyncClient,
    auth_headers_department_head: dict,
    test_risk_with_owner: dict,  # Risk with owner set, to test owner_id change
):
    """Verify that two EDIT requests for same resource return 409."""
    risk_id = test_risk_with_owner["id"]
    
    # First edit request (change owner)
    response1 = await async_client.patch(
        f"/api/v1/risks/{risk_id}",
        json={"owner_id": None},  # Clearing owner is sensitive
        headers=auth_headers_department_head,
    )
    assert response1.status_code == 202
    
    # Second edit request (different field, but still an EDIT action)
    response2 = await async_client.patch(
        f"/api/v1/risks/{risk_id}",
        json={"department_id": 999},  # Different field
        headers=auth_headers_department_head,
    )
    # Should be blocked by existing pending edit OR by our app-layer check
    assert response2.status_code in [400, 409]
```
