Use Python's `ipaddress` module:

```python
import ipaddress
from typing import List

def is_trusted_proxy(client_ip: str, trusted_proxies: List[str]) -> bool:
    """
    Check if client_ip is within any of the trusted proxy CIDR ranges.
    
    Args:
        client_ip: The IP address to check (e.g., "10.0.0.5")
        trusted_proxies: List of CIDR strings (e.g., ["10.0.0.0/8", "192.168.0.0/16"])
    
    Returns:
        True if client_ip is in any trusted network
    """
    try:
        ip = ipaddress.ip_address(client_ip)
    except ValueError:
        return False  # Invalid IP format
    
    for proxy in trusted_proxies:
        try:
            network = ipaddress.ip_network(proxy, strict=False)
            if ip in network:
                return True
        except ValueError:
            continue  # Invalid CIDR format, skip
    
    return False
```

Handle edge cases:

- IPv4 vs IPv6 (don't mix)
- Single IPs without CIDR notation (treat as /32)
- Invalid format strings (skip with warning)

```python
import pytest
from app.middleware.security import is_trusted_proxy

class TestCIDRMatching:
    def test_ipv4_in_cidr_range(self):
        assert is_trusted_proxy("10.0.0.5", ["10.0.0.0/8"])
        assert is_trusted_proxy("10.255.255.255", ["10.0.0.0/8"])
    
    def test_ipv4_not_in_cidr_range(self):
        assert not is_trusted_proxy("11.0.0.1", ["10.0.0.0/8"])
    
    def test_single_ip_without_cidr(self):
        assert is_trusted_proxy("192.168.1.1", ["192.168.1.1/32"])
        assert is_trusted_proxy("192.168.1.1", ["192.168.1.1"])
    
    def test_multiple_networks(self):
        trusted = ["10.0.0.0/8", "192.168.0.0/16", "172.16.0.0/12"]
        assert is_trusted_proxy("10.1.2.3", trusted)
        assert is_trusted_proxy("192.168.100.50", trusted)
        assert is_trusted_proxy("172.20.0.1", trusted)
        assert not is_trusted_proxy("8.8.8.8", trusted)
    
    def test_ipv6_cidr(self):
        assert is_trusted_proxy("2001:db8::1", ["2001:db8::/32"])
        assert not is_trusted_proxy("2001:db9::1", ["2001:db8::/32"])
    
    def test_invalid_ip_format(self):
        assert not is_trusted_proxy("not-an-ip", ["10.0.0.0/8"])
        assert not is_trusted_proxy("", ["10.0.0.0/8"])
    
    def test_invalid_cidr_format(self):
        # Invalid CIDR should be skipped, not crash
        assert not is_trusted_proxy("10.0.0.1", ["invalid-cidr"])
```
