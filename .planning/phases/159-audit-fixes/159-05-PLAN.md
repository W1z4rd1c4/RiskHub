In this codebase, `pending_changes` is typically shaped as:
`{field_name: {"old": ..., "new": ...}}` and the execution path currently does:
`setattr(model, field, vals.get("new"))` for any `hasattr(model, field)`.

This allows writing to ANY model attribute, including:

- `id` (primary key)

```python
# Editable fields per resource type
EDITABLE_FIELDS = {
    "risk": {
        "name", "description", "process", "category", "risk_type",
        "gross_probability", "gross_impact", "net_probability", "net_impact",
        "owner_id", "department_id", "status", "mitigations",
    },
    "control": {
        "name", "description", "frequency", "control_type", "effectiveness",
        "owner_id", "department_id", "is_active",
    },
    "kri": {
        "name", "description", "threshold_green", "threshold_amber", "threshold_red",
        "current_value", "target_value", "reporting_owner_id",
    },
}

def apply_pending_changes(entity, pending_changes: dict, resource_type: str):
    """Apply only whitelisted fields from pending_changes."""
    allowed = EDITABLE_FIELDS.get(resource_type, set())
    
    rejected_fields: list[str] = []
    for field, value in pending_changes.items():
        if field not in allowed:
            rejected_fields.append(field)
            continue
        if hasattr(entity, field):
            # pending_changes values are usually {"old": ..., "new": ...}
            new_value = value.get("new") if isinstance(value, dict) else value
            setattr(entity, field, new_value)

    # Emit ONE audit log event per approval with rejected field names (no values).
    # (See Phase 159 context: Security Logging + Backward Compatibility)
```

Backward compatibility rule (from Phase 159 context): filter at execution time:
- Apply allowed keys.
- Ignore disallowed keys.
- Emit a single audit log event listing rejected field names (no values).

KRI note: preserve the special non-field keys used by KRI flows (`history_entry_id`, `period_end`, `recorded_at`, etc.) by handling them explicitly rather than relying on `hasattr`.

```python
import pytest
from app.services.approval_execution_service import apply_pending_changes

class TestApprovalFieldWhitelist:

    def test_whitelisted_field_applied(self, mock_risk_entity):
        """Whitelisted fields should be set."""
        changes = {"name": "New Name", "description": "New Desc"}
        apply_pending_changes(mock_risk_entity, changes, "risk")
        assert mock_risk_entity.name == "New Name"
    
    def test_non_whitelisted_field_rejected(self, mock_risk_entity):
        """Non-whitelisted fields should be ignored."""
        original_id = mock_risk_entity.id
        original_created_at = mock_risk_entity.created_at
        
        changes = {
            "name": {"old": mock_risk_entity.name, "new": "New Name"},
            "id": {"old": original_id, "new": 999999},  # Attempt to change primary key
            "created_at": {"old": original_created_at, "new": "2020-01-01T00:00:00Z"},  # Attempt to change audit field
        }
        apply_pending_changes(mock_risk_entity, changes, "risk")
        
        assert mock_risk_entity.name == "New Name"  # Allowed
        assert mock_risk_entity.id == original_id  # Blocked
        assert mock_risk_entity.created_at == original_created_at  # Blocked
    
    def test_unknown_resource_type_rejects_all(self, mock_entity):
        """Unknown resource types should reject all fields."""
        changes = {"name": "Hack"}
        apply_pending_changes(mock_entity, changes, "unknown_type")
        # Should not modify anything if resource type has no whitelist
```
